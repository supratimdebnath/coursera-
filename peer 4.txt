char a;
int led=13;

void setup() {
pinMode(led,OUTPUT);
Serial.begin(9600);
}

void loop() { 
digitalWrite(led,LOW);
if(Serial.available()>0)
{
  a=Serial.read();
  if(a=='1')
  {
    Serial.println(a);
    digitalWrite(led,HIGH);
    delay(200);    
  }
  if(a=='0')
  {
    Serial.println(a);
    digitalWrite(led,LOW);
    delay(200);
  }
  
}  // put your main code here, to run repeatedly:

}
